pipeline {
    agent any

    environment {
        DOCKER_IMAGE = "abctechnologies-web:latest"
    }

    stages {
        stage('Clone Repository') {
            steps {
                echo 'Cloning source code from GitHub...'
                // Checked out automatically by Jenkins if using a Pipeline project linked to Git
            }
        }

        stage('Build Docker Image') {
            steps {
                echo 'Building the Docker image...'
                sh "docker build -t ${DOCKER_IMAGE} ."
            }
        }

        stage('Deploy to Kubernetes') {
            steps {
                echo 'Deploying the container to Kubernetes cluster...'
                // Applies the Kubernetes configurations
                sh "kubectl apply -f k8s-deployment.yaml"
                sh "kubectl apply -f k8s-service.yaml"
            }
        }
    }
}
